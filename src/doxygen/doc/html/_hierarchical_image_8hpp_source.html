51 {
52  using namespace std;
53  namespace bf = boost::filesystem3;
54 
55  try {
56  if(!write_image_head_file(file_name)) return false;
57 
58  /* set the img_data_path */
59  set_image_data_path(file_name);
60 
61  /* check the validation of img_data_path */
62  bf::path data_path(img_data_path);
63  if(bf::exists(data_path)) {
64  bf::remove_all(data_path);
65  cout << "[Warning] : " << data_path.generic_string()
66  << " is existing, and the original directory will be removed" << endl;
67  }
68  if(!bf::create_directory(data_path)) {
69  cerr << "create directory " << data_path.generic_string() << " failure" << endl;
70  return false;
71  }
72 
73  const int64 file_number = std::ceil((double)(img_container.size()) / file_node_size);
74 
75  /* now write the code for multiply ways concurrently writing image data */
76  size_t max_concurrent_loop = (get_max_image_level() + 1) / concurrent_number;
77  for(size_t concurrent_loop = 0; concurrent_loop < max_concurrent_loop; ++concurrent_loop) {
78  size_t start_level = concurrent_loop*concurrent_number;
79 
80  /* write concurrent_number level in concurrent begging from the start_level */
81  if(!write_image_inner_loop(start_level, concurrent_number, data_path, file_number))
82  return false;
83  } // end for concurrent loop
84 
85  /* write the residual concurrent_loop */
86  size_t start_level = max_concurrent_loop * concurrent_number;
87  if(!write_image_inner_loop(start_level, get_max_image_level() - start_level + 1, data_path, file_number))
88  return false;
89 
90  /* save the mini image as a jpg format */
91  if(!save_mini_image(file_name)) return false;
92 
93  } catch(bf::filesystem_error &err) {
94  cerr << err.what() << endl;
95  return false;
96  }

102 template<typename T, size_t memory_usage>
103 bool HierarchicalImage<T, memory_usage>::write_image_inner_loop(size_t start_level, size_t merge_number,
104  const boost::filesystem3::path &data_path, const int64 &file_number)
105 {
106  /*
107  * fout_array : different ofstream for multiply ways writing into files
108  * fout_complete : tells whether the ofstream of a file has complete the writing
109  * the file name of ofstream is decided by fout_file_level_number and fout_level_path
110  * for example :
111  * write file into ./some image
112  * then fout_level_path saves like ./some image/level_0 ./some image/level_1 etc
113  * fout_file_level_number saves the current number of each level
114  * the entire file name is like:
115  * ./some image/level_0/fout_file_level_number;
116  */
117 
118  using boost::lexical_cast;
119  using namespace std;
120  namespace bf = boost::filesystem3;
121 
122  static const ContainerType &c_img_container = img_container;
123 
124  std::vector<ofstream> fout_array(merge_number);
125  std::vector<bool> fout_complete(merge_number, true);
126  std::vector<size_t> fout_file_level_number(merge_number, 0);
127  std::vector<bf::path> fout_level_path(merge_number);
128  std::vector<size_t> mask(merge_number, 0);
129 
130  /* initialization */
131  for(size_t k = 0; k < merge_number; ++k) {
132  fout_level_path[k] = data_path / (string("level_") + lexical_cast<std::string>(k+start_level));
133 
134  /* mask is 4^n - 1, start_level is the level delta from last loop */
135  mask[k] = (1 << (2*(k + start_level))) - 1;
136  }
137 
138  /* Before writing, first create the needed directories */
139  for(size_t k = 0; k < merge_number; ++k) {
140  if(!bf::create_directory(fout_level_path[k])) {
141  cerr << "create directory " << fout_level_path[k].generic_string() << " failure" << endl;
142  return false;
143  }
144  }
145 
146  /* writing the first (file_number - 1) data files */
147  int64 file_loop = 0, start_index = 0;
148  for(; file_loop < (file_number - 1); ++file_loop) {
149  /* check ofstream status to decide to whether to open a new file for writing */
150  for(size_t k = 0; k < merge_number; ++k) {
151  if(fout_complete[k]) {
152  string level_file_name =
153  (fout_level_path[k] / lexical_cast<string>(fout_file_level_number[k])).generic_string();
154 
155  /* open file for writing */
156  fout_array[k].open(level_file_name, ios::out | ios::binary);
157  if(!fout_array[k].is_open()) {
158  cerr << "open file " << level_file_name << " failure" << endl;
159  return false;
160  }
161 
162  /* tells the ofstream is in using, thus not complete for creating a new file */
163  fout_complete[k] = false;
164  }
165  }
166 
167  /* write one file data */
168  start_index = file_loop << file_node_shift_num;
169  for(int64 i = 0; i < file_node_size; ++i) {
170  for(size_t k = 0; k < merge_number; ++k) {
171  /* if index i is the multiply of 2^k, then write the data into the fout_array[k] */
172  if((i & mask[k]) == 0)
173  fout_array[k].write(reinterpret_cast<const char*>(&c_img_container[start_index + i]), sizeof(T));
174  }
175  }
176 
177  /* check ofstream status again to decide whether to close a file for writing */
178  for(size_t k = 0; k < merge_number; ++k) {
179  /* attention : using file_loop+1, because the file is counted from 1 */
180  /* if just using file_loop, when file_loop is 0, all the ofstream will be closed that's certainly not correct */
181  if(((file_loop+1) & mask[k]) == 0) {
182  fout_array[k].close();
183 
184  /* prepare for next file writing */
185  fout_file_level_number[k]++;
186  fout_complete[k] = true;
187  }
188  }
189  } // end loop for writing of first (file_number - 1) files
190 
191  /* now just has left one file for writing */
192 
193  /* still first checks whether to open a new file for each ofstream */
194  for(size_t k = 0; k < merge_number; ++k) {
195  if(fout_complete[k]) {
196  string level_file_name = (fout_level_path[k] / lexical_cast<string>(fout_file_level_number[k])).generic_string();
197  fout_array[k].open(level_file_name, ios::out | ios::binary);
198 
199  if(!fout_array[k].is_open()) {
200  cerr << "open file " << level_file_name << " failure" << endl;
201  return false;
202  }
203  }
204  }
205 
206  /* write the file data */
207  start_index = file_loop << file_node_shift_num;
208  for(int64 last_index = start_index; last_index < c_img_container.size(); ++last_index) {
209  for(size_t k = 0; k < merge_number; ++k) {
210  /* if index last_index is the multiply of 2^k, then write the data into the fout_array[k] */
211  if((last_index & mask[k]) == 0)
212  fout_array[k].write(reinterpret_cast<const char*>(&c_img_container[last_index]), sizeof(T));
213  }
214  }
215 
216  /* now just close all the files */
217  for(size_t k = 0; k < merge_number; ++k)
218  fout_array[k].close();
219 
220  return true;
221 }
