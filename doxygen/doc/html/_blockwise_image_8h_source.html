165  size_t rows, size_t cols, size_t mini_rows, size_t mini_cols,
166  boost::shared_ptr<IndexMethodInterface> method = boost::shared_ptr<IndexMethodInterface>())
167 {
168  /* ensure memory_usage is bigger than 8 */
169  memory_usage = (memory_usage < 8) ? 8 : memory_usage;
170 
171  /* make memory_usage = 2^order (order >=3 && order <= 12) */
172  while(memory_usage & (memory_usage - 1)) {
173  memory_usage &= (memory_usage - 1);
174  }
175 
176  switch(memory_usage)
177  {
178  case 8:
179  return boost::make_shared<BlockwiseImage<T, 8> >(rows, cols, mini_rows, mini_cols, method);
180  case 16:
181  return boost::make_shared<BlockwiseImage<T, 16> >(rows, cols, mini_rows, mini_cols, method);
182  case 32:
183  return boost::make_shared<BlockwiseImage<T, 32> >(rows, cols, mini_rows, mini_cols, method);
184  case 64:
