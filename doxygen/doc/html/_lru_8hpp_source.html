98  int put_into_lru(const std::string &file_name) {
99  using namespace std;
100 
101  int index = find(file_name);
102 
103  /* value is in the lru caches, just return the index */
104  if(index != npos) {
105  update_count(index);
106  return index;
107  }
108 
109  ifstream fin(file_name, ios::in | ios::binary);
110  if(!fin.is_open()) {
111  cerr << "can't open file " << file_name << endl;
112  return npos;
113  }
114 
115  /* the data cache is not full */
116  if(current_used < file_cache_numbers) {
117  lru_data.push_back(ValueType(file_name, file_cell_numbers));
118  index = current_used++;
119  } else { /* remove one of the last not used data in cache */
120  int max_number = -1;
121 
122  /* the last number count is the most not used data */
123  for(size_t i = 0; i < file_cache_numbers; ++i) {
124  if(lru_data[i].count > max_number) {
125  max_number = lru_data[i].count;
126  index = i;
127  }
128  }
129 
130  if(!write_back_data(index)) return npos;
131 
132  /* put the new data in the remove index, the data will be covered by the new data read from file */
133  lru_data[index].image_file_name = file_name;
134  }
135 
136  /* read the data into cache */
137  std::vector<T> &data = lru_data[index].image_data;
138  fin.read(reinterpret_cast<char*>(&data[0]), file_cell_numbers*sizeof(T));
139  if(!fin.eof() && fin.fail()) {
140  cerr << "read image file " << lru_data[index].image_file_name << " fails" << endl;
141  return npos;
142  }
143 
144  fin.close();
145  update_count(index);
146  return index;
147  }
